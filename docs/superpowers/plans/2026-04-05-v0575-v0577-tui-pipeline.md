```python
@dataclass
class PipelineSnapshot:
    """Classified task lists for each pipeline stage."""
    building: list[dict]         # status=active, not review task
    backlog: list[dict]          # status=ready, no superseded attempts
    awaiting_review: list[dict]  # status=done, no verdict, not review task
    under_review: list[dict]     # review-* tasks with status=active
    merge_ready: list[dict]      # passing verdict, no Soldier block reason
    merge_blocked: list[dict]    # passing verdict, Soldier wrote block reason
    kicked_back: list[dict]      # status=ready, has superseded attempt
    recently_merged: list[dict]  # has merged attempt (last N)
    review_tasks: dict           # review-{task_id}-{attempt_id} → task dict


def _classify_tasks(self, tasks: list[dict]) -> PipelineSnapshot:
    """Classify all tasks into pipeline stages. Returns PipelineSnapshot."""
    snap = PipelineSnapshot(
        building=[], backlog=[], awaiting_review=[], under_review=[],
        merge_ready=[], merge_blocked=[], kicked_back=[],
        recently_merged=[], review_tasks={},
    )

    for t in tasks:
        tid = t.get("id", "")
        s = t.get("status", "")
        is_review = tid.startswith("review-")

        if is_review:
            snap.review_tasks[tid] = t
            if s == "active":
                snap.under_review.append(t)
            continue

        if s == "active":
            snap.building.append(t)
        elif s == "ready":
            if self._is_kicked_back(t):
                snap.kicked_back.append(t)
            else:
                snap.backlog.append(t)
        elif s == "done":
            verdict = self._get_verdict(t)
            if verdict and verdict.get("verdict") == "pass":
                # Use Soldier-produced block reason (not TUI heuristics)
                block_reason = self._get_merge_block_reason(t)
                if block_reason:
                    snap.merge_blocked.append(t)
                else:
                    snap.merge_ready.append(t)
            else:
                snap.awaiting_review.append(t)
        if self._has_merged_attempt(t):
            snap.recently_merged.append(t)

    return snap
```

```python
def _is_kicked_back(self, task: dict) -> bool:
    """Check if a ready task was kicked back (has superseded attempt)."""
    attempts = task.get("attempts", [])
    return any(a.get("status") == "superseded" for a in attempts)

def _get_verdict(self, task: dict) -> dict | None:
    """Get review verdict from current attempt."""
    current = task.get("current_attempt")
    if not current:
        return None
    for a in task.get("attempts", []):
        if a.get("attempt_id") == current:
            return a.get("review_verdict")
    return None

def _has_merged_attempt(self, task: dict) -> bool:
    """Check if task has any merged attempt."""
    return any(a.get("status") == "merged" for a in task.get("attempts", []))

def _get_merge_block_reason(self, task: dict) -> str | None:
```

```python
def _render_pipeline_bar(self, counts: dict, width: int = 50) -> Text:
    """Render colored pipeline distribution bar."""
    total = sum(counts.values()) or 1
    segments = []
    colors = {
        "bklg": "blue", "bld": "yellow", "rev": "cyan",
        "mrg": "green", "merged": "dim",
    }
    for key, count in counts.items():
        chars = max(1, int(count / total * width)) if count > 0 else 0
        segments.append((colors.get(key, "white"), "█" * chars))

    text = Text()
    for color, block in segments:
        text.append(block, style=color)
    return text
```

```python
def get_app(backend, data_dir=".antfarm", auth_secret=None,
            enable_soldier=True) -> FastAPI:
    # ... existing setup ...

    if enable_soldier:
        _start_soldier_thread(backend, data_dir)

    return app


_soldier_status = "disabled"
_soldier_thread = None  # singleton guard

def _start_soldier_thread(backend, data_dir):
    """Start Soldier as a singleton daemon thread inside the colony process.

    Invariants:
    - Only one Soldier loop per colony process (singleton)
    - Review-task creation is idempotent (checks before creating)
    - Restart-safe: resumes from current state, no duplicate merges
    - Health surfaced in /status/full as soldier_status
    """
    global _soldier_status, _soldier_thread
    if _soldier_thread is not None:
        return  # already running — singleton

    import threading
    from antfarm.core.soldier import Soldier

    def _soldier_loop():
        global _soldier_status
        # Soldier operates directly on the backend — no HTTP loopback.
        # This avoids hardcoding localhost/port and auth issues.
        soldier = Soldier.from_backend(
            backend=backend,
            repo_path=data_dir,
        )
        _soldier_status = "active"
        try:
            while True:
                try:
                    soldier.process_done_tasks()  # idempotent review creation
                    queue = soldier.get_merge_queue()
                    for task in queue:
                        soldier.attempt_merge(task)
                except Exception as exc:
                    logger.warning("soldier loop error: %s", exc)
                    _soldier_status = f"error: {exc}"
                time.sleep(30)
                _soldier_status = "active"
        except Exception as exc:
            _soldier_status = f"crashed: {exc}"

    _soldier_thread = threading.Thread(target=_soldier_loop, daemon=True,
                                        name="soldier")
    _soldier_thread.start()
```

```python
def _build_artifact(self, task: dict, attempt_id: str,
                    workspace: str, branch: str) -> dict:
    """Build TaskArtifact dict from git state after agent completes."""
    import subprocess

    artifact = {
        "task_id": task["id"],
        "attempt_id": attempt_id,
        "worker_id": self.worker_id,
        "branch": branch,
        "pr_url": None,
        "files_changed": [],
        "lines_added": 0,
        "lines_removed": 0,
        "base_commit_sha": "",
        "head_commit_sha": "",
        "target_branch": self.workspace_mgr.integration_branch,
        "target_branch_sha_at_harvest": "",
        "build_ran": False,
        "build_passed": None,
        "tests_ran": False,
        "tests_passed": None,
        "lint_ran": False,
        "lint_passed": None,
        "verification_commands": [],
        "merge_readiness": "needs_review",
        "blocking_reasons": [],
    }

    # Get diff stats
    r = subprocess.run(
        ["git", "diff", "--stat", f"origin/{self.workspace_mgr.integration_branch}...HEAD"],
        cwd=workspace, capture_output=True, text=True, check=False,
    )
    if r.returncode == 0:
        # Parse files from diff stat output
        for line in r.stdout.strip().splitlines()[:-1]:  # skip summary line
            fname = line.split("|")[0].strip()
            if fname:
                artifact["files_changed"].append(fname)

    # Get numstat for lines added/removed
    r = subprocess.run(
        ["git", "diff", "--numstat", f"origin/{self.workspace_mgr.integration_branch}...HEAD"],
        cwd=workspace, capture_output=True, text=True, check=False,
    )
    if r.returncode == 0:
        for line in r.stdout.strip().splitlines():
            parts = line.split("\t")
            if len(parts) >= 2:
                try:
                    artifact["lines_added"] += int(parts[0])
                    artifact["lines_removed"] += int(parts[1])
                except ValueError:
                    pass

    # Get SHAs
    r = subprocess.run(
        ["git", "rev-parse", "HEAD"],
        cwd=workspace, capture_output=True, text=True, check=False,
    )
    if r.returncode == 0:
        artifact["head_commit_sha"] = r.stdout.strip()

    r = subprocess.run(
        ["git", "rev-parse", f"origin/{self.workspace_mgr.integration_branch}"],
        cwd=workspace, capture_output=True, text=True, check=False,
    )
    if r.returncode == 0:
        artifact["target_branch_sha_at_harvest"] = r.stdout.strip()

    # Get base commit SHA (merge-base between HEAD and integration branch)
    r = subprocess.run(
        ["git", "merge-base", "HEAD",
         f"origin/{self.workspace_mgr.integration_branch}"],
        cwd=workspace, capture_output=True, text=True, check=False,
    )
    if r.returncode == 0:
        artifact["base_commit_sha"] = r.stdout.strip()

    return artifact
```

```python
def _create_pr(self, task: dict, branch: str, workspace: str) -> str:
    """Create a GitHub PR for the completed work. Returns PR URL or ""."""
    title = task.get("title", task["id"])
    body = f"Antfarm task: {task['id']}\n\nWorker: {self.worker_id}"

    r = subprocess.run(
        ["gh", "pr", "create", "--title", title, "--body", body,
         "--head", branch, "--fill"],
        cwd=workspace, capture_output=True, text=True, check=False,
    )
    if r.returncode == 0:
        return r.stdout.strip()  # PR URL
    return ""
```

### 5. Worker announces exit (#101)

```python
def run(self) -> None:
    # ... existing setup ...
    try:
        while True:
            had_task = self._process_one_task()
            if not had_task:
                logger.info("queue empty, worker exiting")
                break
    finally:
        # Announce exit before deregistering
        # Explicit offline transition — distinguishable from crash
        # Trail on last task if one was processed
        if self._last_task_id:
            try:
                self.colony.trail(
                    self._last_task_id, self.worker_id,
                    "worker exiting — queue empty",
                    action_type="worker_exit",
                )
            except Exception:
                pass

        # Always: explicit heartbeat with status="offline" before deregister.
        # This lets the TUI distinguish intentional exit from crash
        # (crash = heartbeat goes stale; exit = status explicitly set to offline)
        try:
            self.colony.heartbeat(self.worker_id, status={"status": "offline"})
        except Exception:
            pass
        self.colony.deregister_worker(self.worker_id)
```

```python
import collections
_event_queue: collections.deque = collections.deque(maxlen=1000)
_event_counter = 0  # monotonic counter for cursor-based consumption

@app.get("/events")
def event_stream(timeout: float = Query(default=0.0)):
    """SSE stream for task state change events.

    Informational only — visibility, not truth. System correctness
    must not depend on SSE delivery.
    """
    def _generate():
        seen = _event_counter
        start = time.monotonic()
        while True:
            if timeout > 0 and (time.monotonic() - start) >= timeout:
                break
            # Iterate new events since last seen
            current = list(_event_queue)  # snapshot
            for event in current:
                yield f"data: {json.dumps(event)}\n\n"
            seen = len(_event_queue)
            time.sleep(1)
    return StreamingResponse(_generate(), media_type="text/event-stream")
```

2. Clean `.antfarm/` data directory
3. Restart colony with v0.5.76 code: `antfarm colony --port 7433`

   - `antfarm worker start --type builder --node mini-2 --agent claude-code ...`
7. Start reviewer worker:
   - `antfarm worker start --type reviewer --node mini-2 --agent claude-code ...`
8. Open TUI: `antfarm scout --tui`

10. Verify:
    - [ ] Artifacts exist on all done tasks
    - [ ] PRs created on GitHub
    - [ ] Trail entries visible in TUI
    - [ ] Review tasks created by Soldier
    - [ ] Reviewer produces verdicts
    - [ ] Soldier merges on passing verdict
    - [ ] Kicked-back tasks visible with findings
    - [ ] Workers announce exit when queue empties
    - [ ] Node names + worker types visible in summary
    - [ ] Progress bar + pipeline bar accurate

### Success criteria

- Tasks flow through the entire pipeline without manual intervention
- The TUI shows every stage accurately
- No silent failures — every state change is visible
- No task remains in Awaiting Review longer than 60s without a visible reason
- No duplicate review tasks or duplicate merges during restart/recovery
- Worker exit is distinguishable from worker crash in the TUI
